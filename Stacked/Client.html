def configure
  yield self

  @site     ||= DEFAULT_OPTIONS[:site]
  @version  ||= DEFAULT_OPTIONS[:version]
  @api_key  ||= DEFAULT_OPTIONS[:api_key]
  @base_url ||= "http://api.#{@site}/#{@version}/"

  true
end
